interface Ethernet1
   no switchport
   ip address 192.168.1.2/30
!
interface Ethernet2
   no switchport
   ip address 192.168.2.2/30
!
interface Ethernet3
   switchport access vlan 10
!
interface Ethernet4
   switchport access vlan 20
!
interface Loopback1
   ip address 11.11.11.11/32
!
interface Vlan10
   vrf t1
   ip address 192.168.1.50/24
!
interface Vlan20
   vrf t1
   ip address 192.168.2.50/24
!
interface Vxlan1
   vxlan source-interface Loopback1
   vxlan controller-client
   vxlan udp-port 4789
   vxlan vlan 10 vni 100
   vxlan vlan 20 vni 200
!
ip routing
ip routing vrf t1
!
ip route 1.1.1.1/32 192.168.1.1
ip route 2.2.2.2/32 192.168.2.1
!
router bgp 11
   neighbor SPINE peer group
   neighbor SPINE remote-as 1
   neighbor SPINE update-source Loopback1
   neighbor SPINE ebgp-multihop 2
   neighbor SPINE send-community
   neighbor 1.1.1.1 peer group SPINE
   neighbor 2.2.2.2 peer group SPINE
   !
   vlan 10
      rd 10:2
      route-target both 10:10
      redistribute learned
   !
   vlan 20
      rd 20:2
      route-target both 20:20
      redistribute learned
   !
   address-family evpn
      neighbor SPINE activate
   !
   address-family ipv4
      network 11.11.11.11/32
!
end
